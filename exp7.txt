🔹 EXPERIMENT: AST GENERATION USING YACC
✅ AIM

To convert BNF rules into YACC grammar and to generate an Abstract Syntax Tree (AST) for arithmetic expressions using syntax-directed translation in YACC.

✅ ALGORITHM

Step 1: Start
Begin the program execution.

Step 2: Define BNF Grammar
Example BNF:

E → E + E
E → E * E
E → (E)
E → number

Step 3: Convert BNF to YACC Form

Rewrite rules in YACC format
Define precedence:
* > +
Declare tokens

Step 4: Design AST Node Structure

Create node with:
data (operator/operand)
left pointer
right pointer

Step 5: Lexical Analysis

Identify numbers, operators, parentheses
Send tokens to YACC

Step 6: Syntax Analysis & AST Construction

For each rule:
Create node
Attach children

Step 7: Store AST Root

Save final root node

Step 8: Output

Traverse tree (inorder)

Step 9: Stop

🔸 YACC FILE (ast.y)
%{
#include <stdio.h>
#include <stdlib.h>

typedef struct node {
    int data;
    struct node *left;
    struct node *right;
} node;

node* newNode(int data, node* left, node* right) {
    node* temp = (node*)malloc(sizeof(node));
    temp->data = data;
    temp->left = left;
    temp->right = right;
    return temp;
}

void inorder(node* root) {
    if (root != NULL) {
        inorder(root->left);
        if (root->data == '+' || root->data == '*' ||
            root->data == '-' || root->data == '/')
            printf(" %c ", root->data);
        else
            printf(" %d ", root->data);
        inorder(root->right);
    }
}

node* root;

void yyerror(const char *s);
int yylex();
%}

%union {
    int num;
    node* ptr;
}

%token <num> NUMBER
%type <ptr> E

%left '+' '-'
%left '*' '/'

%%

E : E '+' E   { $$ = newNode('+', $1, $3); root = $$; }
  | E '-' E   { $$ = newNode('-', $1, $3); root = $$; }
  | E '*' E   { $$ = newNode('*', $1, $3); root = $$; }
  | E '/' E   { $$ = newNode('/', $1, $3); root = $$; }
  | '(' E ')' { $$ = $2; }
  | NUMBER    { $$ = newNode($1, NULL, NULL); root = $$; }
  ;

%%

int main() {
    printf("Enter expression:\n");
    yyparse();

    printf("\nInorder Traversal of AST:\n");
    inorder(root);

    return 0;
}

void yyerror(const char *s) {
    printf("Invalid Expression\n");
}
